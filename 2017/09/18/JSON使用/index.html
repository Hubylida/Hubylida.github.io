JSON的使用

json是拯救被xml折磨的程序员的一道光。

它是由伟大的红宝书作者，道格拉斯所创造，真是厉害啊，他写的RegExp我是真的不想看。

json有六种数据类型，分别是:

number，boolean，string，null，array，object

在json中必须使用双引号"",Object的键也必须使用双引号"";

把JavaScript的对象序列化就变成json格式的字符串，在反序列化后就变成对象。

序列化

var gakki = {
    name: 'gakki',
    age: '29',
    height: 169 + 'cm',
    weight: 55 + 'kg',
    profession: ''actress,
    husband: 'sifeichai'
}
JSON.stringify(gakki);

//"{"height":"169cm","weight":"55kg","profession":"actress","husband":"sifeichai"}"]

用JSON.stringify(gakki, null,' ');控制缩进

"{
    "name": "gakki",
    "age": "29",
    "height": "169cm",
    "weight": "55kg",
    "profession": "actress",
    "husband": "sifeichai"
}"

第二个参数表示输出对象的键值，可控制输出的属性。以数组的形式。

JSON.stringify(gakki,['height','weight'],' ');

"{
    "height": "169cm",
    "weight": "55kg"
}"

还可以传入一个函数

function up(key,value){
    if(typeof value === 'string'){
        return value.toUpperCase();
    }
    return value;
}
JSON.stringify(gakki,up,'   ');

结果

"{
    "name": "GAKKI",
    "age": "29",
    "height": "169CM",
    "weight": "55KG",
    "profession": "ACTRESS",
    "husband": "SIFEICHAI"
}"

还可以直接给对象一个toJSON的方法，直接返回JSON应该序列化的数

var gakki = {
    name: 'gakki',
    age: '29',
    height: 169 + 'cm',
    weight: 55 + 'kg',
    profession: 'actress',
    husband: 'sifeichai',
    toJSON: function(){
        return {
            'name': this.name,
            'height': this.height
        };
    }
};

JSON.stringify(gakki, null, '   ');

//
"{
    "name": "gakki",
    "height": "169cm"
}"

反序列化

使用方法，JSON.parse() 把json格式的字符串转化为js对象。

JSON.parse('[1,2,3,true]'); // [1, 2, 3, true]
JSON.parse('{"name":"小明","age":14}'); // Object {name: '小明', age: 14}
JSON.parse('true'); // true
JSON.parse('123.45'); // 123.45
